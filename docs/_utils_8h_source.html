330  template<typename T, typename A>
331  double vector_median(std::vector<T, A> data) {
332  if (data.size() % 2 == 0) {
333  const auto median_it1 = data.begin() + data.size() / 2 - 1;
334  const auto median_it2 = data.begin() + data.size() / 2;
335  std::nth_element(data.begin(), median_it1, data.end());
336  T val1 = *median_it1;
337  std::nth_element(data.begin(), median_it2, data.end());
338  T val2 = *median_it2;
339  return (double)(val1 + val2) / 2.0;
340  }
341  else {
342  const auto median_it = data.begin() + data.size() / 2;
343  std::nth_element(data.begin(), median_it, data.end());
344  return *median_it;
345  }
346  }
