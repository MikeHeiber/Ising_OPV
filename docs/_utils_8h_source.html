256  template<typename T>

258  if ((int)vec.size() > 2) {
259  for (auto it1 = vec.begin(); it1 != vec.end() - 2; ++it1) {
260  for (auto it2 = it1 + 2; it2 != vec.end(); ++it2) {
261  if (*it2 == *it1) {
262  *it2 = *(it2 - 1);
263  }
264  }
265  }
266  }

268  auto it3 = unique(vec.begin(), vec.end());
