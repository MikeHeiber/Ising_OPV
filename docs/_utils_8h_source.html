364  template<typename T, typename A>
365  double vector_median(std::vector<T, A> data) {
366  if (data.size() % 2 == 0) {
367  const auto median_it1 = data.begin() + data.size() / 2 - 1;
368  const auto median_it2 = data.begin() + data.size() / 2;
369  std::nth_element(data.begin(), median_it1, data.end());
370  T val1 = *median_it1;
371  std::nth_element(data.begin(), median_it2, data.end());
372  T val2 = *median_it2;
373  return (double)(val1 + val2) / 2.0;
374  }
375  else {
376  const auto median_it = data.begin() + data.size() / 2;
377  std::nth_element(data.begin(), median_it, data.end());
378  return *median_it;
379  }
380  }
