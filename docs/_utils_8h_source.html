362  template<typename T, typename A>
363  double vector_median(std::vector<T, A> data) {
364  if (data.size() % 2 == 0) {
365  const auto median_it1 = data.begin() + data.size() / 2 - 1;
366  const auto median_it2 = data.begin() + data.size() / 2;
367  std::nth_element(data.begin(), median_it1, data.end());
368  T val1 = *median_it1;
369  std::nth_element(data.begin(), median_it2, data.end());
370  T val2 = *median_it2;
371  return (double)(val1 + val2) / 2.0;
372  }
373  else {
374  const auto median_it = data.begin() + data.size() / 2;
375  std::nth_element(data.begin(), median_it, data.end());
376  return *median_it;
377  }
378  }
